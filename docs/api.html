Core API — FastCipher

Creating a Cipher

Rust

use fast_core::{FastCipher, FastKey, Domain, SecurityLevel};

// 16-byte key (AES-128) or 32-byte key (AES-256)
let key = FastKey::new(&[0x42u8; 16])?;

// Choose domain and security level
let cipher = FastCipher::new(
    &key,
    Domain::Decimal,             // radix 10: digits 0-9
    SecurityLevel::Quantum128,   // quantum-resistant parameters
)?;

Python

from fast_fpe import FastCipher

cipher = FastCipher(
    key=b'\x42' * 16,       # 16 or 32 bytes
    radix=10,                # 10=decimal, 36=alphanumeric, 62=mixed-case
    security="quantum-128",  # "classical-128" or "quantum-128"
)

Encrypt & Decrypt

Rust

// Encrypt with a tweak (domain separator)
let ciphertext = cipher.encrypt(b"my-tweak", "123456789")?;
// → "847920134" (9 decimal digits, same format)

// Decrypt with the same tweak
let plaintext = cipher.decrypt(b"my-tweak", &ciphertext)?;
assert_eq!(plaintext, "123456789");

Python

ct = cipher.encrypt(tweak=b"my-tweak", plaintext="123456789")
pt = cipher.decrypt(tweak=b"my-tweak", ciphertext=ct)
assert pt == "123456789"
Tweak must match. Decryption requires the exact same tweak used for encryption. Different tweaks produce different ciphertexts for the same plaintext — this is by design for domain separation.

Batch Operations

For high-throughput tokenization, pre-compute the cipher state once and reuse it. This amortizes the expensive key setup (S-box generation) across millions of operations.

Rust

use fast_core::{FastCipher, FastCipherState, FastKey, Domain, SecurityLevel};

let key = FastKey::new(&[0x42u8; 16])?;

// Pre-compute state once (expensive: S-box generation + round sequence)
let state = FastCipherState::setup(
    &key,
    b"411111",                   // tweak (e.g., BIN for credit cards)
    10,                          // radix
    6,                           // block length
    SecurityLevel::Quantum128,
)?;
let mapping = Domain::Decimal.mapping();

// Encrypt millions — each call is pure table lookups, zero AES
for plaintext in &plaintexts {
    let ct = FastCipher::encrypt_with_state(&state, plaintext, mapping.as_ref())?;
    // process ct...
}

// Decrypt with same state
for ciphertext in &ciphertexts {
    let pt = FastCipher::decrypt_with_state(&state, ciphertext, mapping.as_ref())?;
}

Credit Card Tokenization

Preserve the BIN (first 6) and last 4, encrypt the middle digits with BIN as tweak:

Rust

let key = FastKey::new(&[0x42u8; 16])?;
let cipher = FastCipher::new(&key, Domain::Decimal, SecurityLevel::Quantum128)?;

let pan = "4111111111111111";
let bin = &pan[..6];          // "411111" — preserved
let middle = &pan[6..12];     // "111111" — encrypted
let last4 = &pan[12..];       // "1111" — preserved

let tokenized = cipher.encrypt(bin.as_bytes(), middle)?;
let token = format!("{bin}{tokenized}{last4}");
// → "411111XXXXXX1111" — valid 16-digit PAN

Python

pan = "4111111111111111"
bin_prefix = pan[:6]
middle = pan[6:12]
last4 = pan[12:]

tokenized_middle = cipher.encrypt(tweak=bin_prefix.encode(), plaintext=middle)
token = bin_prefix + tokenized_middle + last4

FF1 Comparison & Migration

FF1 Cipher

# Python
from fast_fpe import Ff1Cipher

ff1 = Ff1Cipher(key=b'\x00' * 16, radix=10)
ct = ff1.encrypt(tweak=b"tweak", plaintext="123456789")
pt = ff1.decrypt(tweak=b"tweak", ciphertext=ct)

FF1 → FAST Migration

use fast_migrate::Ff1ToFastMigrator;

let migrator = Ff1ToFastMigrator::new(
    &ff1_key,                    // existing FF1 key
    &fast_key,                   // new FAST key
    10,                          // radix
    SecurityLevel::Quantum128,
)?;

// Single token
let new_token = migrator.migrate_token(&old_token, &ff1_tweak, &fast_tweak)?;

// Batch with progress
let results = migrator.migrate_batch(&tweak, &tokens, |done, total| {
    println!("Progress: {done}/{total}");
})?;

Security Levels

LevelRounds per PositionTotal Rounds (10-digit)Quantum Resistant
Classical128 ceil(256 / log₂(a)) ~780 Partial (Grover mitigation only)
Quantum128 ceil(384 / log₂(a)) ~1160 Yes
Recommendation: Use Quantum128 for all new deployments. The 50% round increase is a small performance cost for quantum resistance insurance.

Error Handling

Rust — Error Types

// CipherError variants:
CipherError::InvalidLength { expected, got }  // wrong plaintext length for state
CipherError::InvalidCharacter { ch, radix }   // character outside domain
CipherError::RadixTooSmall { min, got }       // radix < 4
CipherError::InputTooShort { min, got }       // block length < 2

// SetupError variants:
SetupError::InvalidKeyLength(len)             // key not 16 or 32 bytes
SetupError::Param(ParamError)                 // invalid parameters

Python — Exceptions

from fast_fpe import InvalidInput

try:
    cipher = FastCipher(key=b'\x00' * 8, radix=10, security="quantum-128")
except InvalidInput as e:
    print(f"Error: {e}")  # "invalid key length: expected 16 or 32 bytes, got 8"
