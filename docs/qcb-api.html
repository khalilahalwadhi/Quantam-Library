QCB-AEAD API

Rust & Python API Reference

Rust API

QcbKey

Represents an AES key with automatic zeroization on drop.

use qcb_core::QcbKey;

// AES-256 (32 bytes)
let key = QcbKey::new(&[0x42; 32]).unwrap();

// AES-128 (16 bytes)
let key = QcbKey::new(&[0x42; 16]).unwrap();

// Invalid — returns Err(QcbError::InvalidKeyLength)
let err = QcbKey::new(&[0x00; 8]);
MethodParametersReturns
QcbKey::new(bytes) &[u8] — 16 or 32 bytes Result<QcbKey, QcbError>

Qcb

The main AEAD cipher. Constructed from a QcbKey.

use qcb_core::{Qcb, QcbKey};

let key = QcbKey::new(&[0x42; 32]).unwrap();
let cipher = Qcb::new(&key);

encrypt

let ciphertext = cipher.encrypt(
    &nonce,           // &[u8] — exactly 12 bytes
    &associated_data, // &[u8] — any length (can be empty)
    &plaintext,       // &[u8] — any length (can be empty)
).unwrap();
// ciphertext length = plaintext length + 16 (tag)
ParameterTypeConstraints
nonce&[u8]Exactly 12 bytes. Must be unique per encryption.
aad&[u8]Any length. Authenticated but not encrypted.
plaintext&[u8]Any length up to (2²⁴ − 1) × 16 bytes.

Returns: Result<Vec<u8>, QcbError> — ciphertext with appended 16-byte authentication tag.

decrypt

let plaintext = cipher.decrypt(
    &nonce,
    &associated_data,
    &ciphertext_with_tag,
).unwrap();

Returns: Result<Vec<u8>, QcbError> — plaintext on success, or QcbError::AuthenticationFailed if the tag doesn't verify.

QcbError

VariantMeaning

CiphertextTooShortCiphertext is shorter than 16 bytes (no tag)

QcbCipher

from qcb_aead import QcbCipher, InvalidInput, AuthenticationError

# Create cipher with AES-256 key
cipher = QcbCipher(key=b"\x42" * 32)

# Also supports AES-128
cipher128 = QcbCipher(key=b"\x42" * 16)

encrypt(nonce, data, aad) → bytes

ciphertext = cipher.encrypt(
    nonce=b"\x00" * 12,
    data=b"secret message",
    aad=b"associated data",
)
# len(ciphertext) == len(data) + 16

decrypt(nonce, ciphertext, aad) → bytes

try:
    plaintext = cipher.decrypt(
        nonce=b"\x00" * 12,
        ciphertext=ciphertext,
        aad=b"associated data",
    )
except AuthenticationError:
    print("Tag mismatch — data was tampered!")

Exceptions

ExceptionBaseWhen
InvalidInputValueErrorBad key length, nonce length, or message too long

Complete Example

from qcb_aead import QcbCipher, AuthenticationError
import os

# Generate random key
key = os.urandom(32)
cipher = QcbCipher(key=key)

# Encrypt
nonce = os.urandom(12)
plaintext = b"Hello, quantum-safe world!"
aad = b"message metadata v1"

ciphertext = cipher.encrypt(nonce=nonce, data=plaintext, aad=aad)
print(f"Ciphertext: {len(ciphertext)} bytes (plaintext: {len(plaintext)} + tag: 16)")

# Decrypt
recovered = cipher.decrypt(nonce=nonce, ciphertext=ciphertext, aad=aad)
assert recovered == plaintext

# Tamper detection
tampered = bytearray(ciphertext)
tampered[0] ^= 0x01
try:
    cipher.decrypt(nonce=nonce, ciphertext=bytes(tampered), aad=aad)
except AuthenticationError:
    print("Tampering detected!")
